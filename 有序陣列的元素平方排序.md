#include <vector>
#include <iostream>

class Solution {
public:
    std::vector<int> sortedSquares(std::vector<int>& nums) {
        int sortedIndex = nums.size() - 1;  // 從最後的索引開始排，即從最大到最小
        int leftIndex = 0, rightIndex = nums.size() - 1;  
        std::vector<int> result(nums.size(), 0);  // 初始化

        while (leftIndex <= rightIndex) { 
            int leftSquare = nums[leftIndex] * nums[leftIndex];  
            int rightSquare = nums[rightIndex] * nums[rightIndex];  

            if (leftSquare > rightSquare) {  
                result[sortedIndex--] = leftSquare;  
                leftIndex++;  
            } else {
                result[sortedIndex--] = rightSquare;  
                rightIndex--;  
            }
        }
        return result; 
    }
};

int main() {
    Solution sol;
    std::vector<int> nums = {-5, -2, 0, 4, 9, 16};  
    std::vector<int> result = sol.sortedSquares(nums);

    std::cout << "平方排序後的陣列: ";
    for (const auto& num : result) {
        std::cout << num << " ";  
    }
    std::cout << "\n";

    return 0;
}
