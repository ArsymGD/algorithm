#include <vector>
#include <iostream>

class Solution {
public:
    int removeElement(std::vector<int>& nums, int val) {
        int slowIndex = 0;  // 慢指標，用於記錄不等於 val 的元素位置

        // 原地修改，fast 指標遍歷整個陣列
        for (int fastIndex = 0; fastIndex < nums.size(); fastIndex++) {
            if (nums[fastIndex] != val) {   
                nums[slowIndex++] = nums[fastIndex];  // 當前元素不等於 val 時，將其移至慢指標位置
            }
        }
        return slowIndex;  
    }
};

int main() {
    Solution sol;
    std::vector<int> nums = {6, 9, 5, 3, 2, 3, 0, 2, 4, 2, 3, 5, 8, 2, 3};  
    int val = 2;

    int newLength = sol.removeElement(nums, val);

    std::cout << "新陣列長度: " << newLength << "\n";
    std::cout << "新陣列: ";

   
    for (int i = 0; i < newLength; i++) {
        std::cout << nums[i] << " ";
    }
    std::cout << "\n";

    return 0;
}
